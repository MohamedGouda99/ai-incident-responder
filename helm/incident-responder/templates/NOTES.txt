No Ingress is configured. You can access the application using port-forward:

  # Frontend
  kubectl port-forward svc/{{ include "incident-responder.fullname" . }}-frontend {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.port }} -n {{ .Release.Namespace }}

  # Backend API
  kubectl port-forward svc/{{ include "incident-responder.fullname" . }}-backend {{ .Values.backend.service.port }}:{{ .Values.backend.service.port }} -n {{ .Release.Namespace }}

  Then open:
    Frontend: http://localhost:{{ .Values.frontend.service.port }}
    Backend:  http://localhost:{{ .Values.backend.service.port }}/api/v1/health

{{- end }}
